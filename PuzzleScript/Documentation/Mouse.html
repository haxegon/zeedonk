function update(){
  if (Mouse.leftheld()){
    if (t<30){
     t++;
    }
  } else if (t>0) {
    t--;
  }
  var mx = Mouse.x;
  var my = Mouse.y;
  Gfx.fillbox(mx-1,0,3,Gfx.screenheight,Col.GREY);
  Gfx.fillbox(0,my-1,Gfx.screenwidth,3,Col.GREY);
  Gfx.fillcircle(mx,my,4+t,Col.WHITE);
}

var a =0;
function update(){
  var w = Mouse.mousewheel;
  if (w>0) a++;
  if (w<0) a--;
  for (i in 0...15){
    var c = Gfx.hsl(20*i+10*Game.time,1.0,0.5);
   Gfx.drawhexagon(Gfx.screenwidthmid,Gfx.screenheightmid,8*i,2*(a*i/180),c);
  }
}
